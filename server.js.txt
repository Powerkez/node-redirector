const express = require('express');
const app = express();

const PORT = process.env.PORT || 3000;
const TARGET_URL = 'https://bet.za.com/out2/normaloffic/index.php';

app.get('*', (req, res) => {
    res.redirect(301, TARGET_URL);
});

app.listen(PORT, () => {
    console.log(`Redirector running on port ${PORT}`);
});
